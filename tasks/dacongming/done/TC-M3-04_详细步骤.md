## Phase 3：编辑器核心逻辑

### 步骤 3.1：创建 `src/composables/useStrategyEditor.ts`
```typescript
import { reactive, computed, ref } from 'vue'
import { backend } from '../api/backend'
import ws from '../utils/websocket'
import { ElMessage } from 'element-plus'

export function useStrategyEditor() {
  const selectedFactors = ref<any[]>([])
  const taskId = ref('')

  const formData = reactive({
    name: '',
    stock_filter: {} as Record<string, any>,
    timing: { type: '', period: 20 },
    risk: { max_position: 0.3, stop_loss: 0.05, stop_profit: 0.1 }
  })

  // 计算 JSON 预览
  const JSONPreview = computed(() => {
    const result: any = {
      name: formData.name || '未命名策略',
      stock_filter: {},
      timing: formData.timing,
      risk: formData.risk
    }
    for (const factor of selectedFactors.value) {
      result.stock_filter[factor.key] = formData.stock_filter[factor.key] || {}
    }
    return JSON.stringify(result, null, 2)
  })

  // 因子树选中事件
  const onFactorCheck = (_: any, checkStatus: any) => {
    // 从所有因子树中收集已选中节点
    const allFactors = [...financialFactors, ...technicalFactors, ...sectorFactors]
    const checkedKeys = checkStatus.checkedKeys
    selectedFactors.value = allFactors.filter(f => checkedKeys.includes(f.key))
    // 初始化默认参数
    for (const factor of selectedFactors.value) {
      if (!formData.stock_filter[factor.key]) {
        const defaults: Record<string, any> = {}
        factor.params.forEach((p: any) => { defaults[p.key] = p.default ?? '' })
        formData.stock_filter[factor.key] = defaults
      }
    }
  }

  // 保存草稿
  const saveDraft = async () => {
    try {
      await backend.post('/strategy/save', {
        name: formData.name,
        config: JSON.parse(JSONPreview)
      })
      ElMessage.success('草稿保存成功')
    } catch (e: any) {
      ElMessage.error(e.message || '保存失败')
    }
  }

  // 提交执行
  const submitStrategy = async () => {
    if (!formData.name) {
      ElMessage.warning('请输入策略名称')
      return
    }
    try {
      const res = await backend.post('/strategy/submit', {
        name: formData.name,
        config: JSON.parse(JSONPreview)
      })
      taskId.value = res.data.task_id
      ElMessage.success(`提交成功，task_id: ${taskId.value}`)

      // 监听回测进度
      ws.on('backtest_progress', (data: any) => {
        if (data.task_id === taskId.value) {
          ElMessage.info(`回测进度: ${data.progress}%`)
        }
      })
    } catch (e: any) {
      ElMessage.error(e.message || '提交失败')
    }
  }

  return { formData, selectedFactors, JSONPreview, onFactorCheck, saveDraft, submitStrategy }
}
```
