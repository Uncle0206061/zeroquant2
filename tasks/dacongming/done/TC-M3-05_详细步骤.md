### 步骤 A2：创建 `src/composables/useBacktest.ts`

```typescript
import { ref, reactive, onMounted, nextTick } from 'vue'
import * as echarts from 'echarts'
import { data as dataApi } from '../api/data'
import { backend as backendApi } from '../api/backend'
import ws from '../utils/websocket'
import { ElMessage } from 'element-plus'

export function useBacktest(equityChartRef: any, drawdownChartRef: any, returnDistChartRef: any) {
  const form = reactive({
    stockCode: '000001.SZ',
    startDate: '2024-01-01',
    endDate: '2024-12-31',
    capital: 1000000
  })
  const running = ref(false)
  const progress = ref(0)
  const result = ref<any>(null)

  let equityChart: echarts.ECharts | null = null
  let drawdownChart: echarts.ECharts | null = null
  let returnDistChart: echarts.ECharts | null = null

  // 初始化图表实例
  const initCharts = () => {
    if (equityChartRef.value) equityChart = echarts.init(equityChartRef.value)
    if (drawdownChartRef.value) drawdownChart = echarts.init(drawdownChartRef.value)
    if (returnDistChartRef.value) returnDistChart = echarts.init(returnDistChartRef.value)
  }

  // 绘制资金曲线
  const drawEquityCurve = (equityData: { date: string; equity: number }[]) => {
    if (!equityChart) return
    equityChart.setOption({
      tooltip: { trigger: 'axis' },
      grid: { left: 60, right: 20, top: 20, bottom: 40 },
      xAxis: { type: 'category', data: equityData.map(d => d.date) },
      yAxis: { type: 'value' },
      series: [{
        name: '资金',
        type: 'line',
        data: equityData.map(d => d.equity),
        smooth: true,
        lineStyle: { width: 2 },
        areaStyle: { opacity: 0.2 }
      }]
    })
  }

  // 绘制回撤曲线
  const drawDrawdown = (data: { date: string; drawdown: number }[]) => {
    if (!drawdownChart) return
    drawdownChart.setOption({
      tooltip: { trigger: 'axis' },
      grid: { left: 60, right: 20, top: 20, bottom: 40 },
      xAxis: { type: 'category', data: data.map(d => d.date) },
      yAxis: { type: 'value', axisLabel: { formatter: v => (v * 100).toFixed(1) + '%' } },
      series: [{
        name: '回撤',
        type: 'line',
        data: data.map(d => d.drawdown),
        smooth: true,
        lineStyle: { width: 2, color: '#67C23A' },
        areaStyle: { color: 'rgba(103,194,58,0.1)' }
      }]
    })
  }

  // 绘制收益分布
  const drawReturnDist = (returns: number[]) => {
    if (!returnDistChart) return
    returnDistChart.setOption({
      tooltip: { trigger: 'axis' },
      grid: { left: 60, right: 20, top: 20, bottom: 40 },
      xAxis: { type: 'category' },
      yAxis: { type: 'value' },
      series: [{
        name: '收益',
        type: 'bar',
        data: returns.map(r => ({ value: r, itemStyle: { color: r >= 0 ? '#F56C6C' : '#67C23A' } }))
      }]
    })
  }

  // 发起回测
  const runBacktest = async () => {
    running.value = true
    progress.value = 0

    try {
      // 1. 调用后端回测接口
      const res = await backendApi.post('/strategy/backtest/submit', {
        stock_code: form.stockCode,
        start_date: form.startDate,
        end_date: form.endDate,
        initial_capital: form.capital
      })
      const taskId = res.data.task_id

      // 2. 监听 WebSocket 回测进度
      ws.on('backtest_progress', (d: any) => {
        if (d.task_id === taskId) {
          progress.value = d.progress || 0
        }
      })

      // 3. 等待回测完成（轮询或 WebSocket 结果）
      const resultRes = await backendApi.get(`/strategy/backtest/result/${taskId}`)
      result.value = resultRes.data

      // 4. 渲染图表
      await nextTick()
      initCharts()
      if (result.value.equity_curve) drawEquityCurve(result.value.equity_curve)
      if (result.value.drawdown_curve) drawDrawdown(result.value.drawdown_curve)
      if (result.value.return_distribution) drawReturnDist(result.value.return_distribution)

    } catch (e: any) {
      ElMessage.error(e.message || '回测失败')
    } finally {
      running.value = false
    }
  }

  return { form, running, progress, result, runBacktest }
}
```

## Part B：消息提醒系统

### 步骤 B1：创建全局通知组件 `src/components/NotificationCenter.vue`

```vue
<template>
  <div class="notification-center">
    <!-- 铃铛图标 -->
    <el-badge :value="unreadCount" :hidden="unreadCount === 0" :max="99">
      <el-icon :size="22" @click="togglePanel" class="bell-icon" :class="{ active: unreadCount > 0 }">
        <Bell />
      </el-icon>
    </el-badge>

    <!-- 通知面板（点击铃铛弹出） -->
    <el-popover placement="bottom-end" :width="360" trigger="click" v-model:visible="showPanel">
      <template #reference>
        <span />
      </template>
      <div class="notif-panel">
        <div class="notif-header">
          <span>消息通知</span>
          <el-button link size="small" @click="clearAll">全部清除</el-button>
        </div>
        <el-divider class="notif-divider" />
        <div class="notif-list" v-if="notifications.length > 0">
          <div
            v-for="n in notifications"
            :key="n.id"
            class="notif-item"
            :class="{ unread: !n.read }"
            @click="onNotifClick(n)"
          >
            <el-tag :type="n.typeMap[n.type]" size="small">{{ n.typeLabel }}</el-tag>
            <div class="notif-content">{{ n.content }}</div>
            <div class="notif-time">{{ formatTime(n.time) }}</div>
          </div>
        </div>
        <el-empty v-else description="暂无消息" :image-size="60" />
      </div>
    </el-popover>

    <!-- 右下角浮窗提醒 -->
    <transition name="slide-up">
      <div v-if="toast" class="toast-popup" :class="toast.type">
        <el-icon v-if="toast.type === 'success'" color="#67C23A"><Check /></el-icon>
        <el-icon v-if="toast.type === 'warning'" color="#E6A23C"><Warning /></el-icon>
        <el-icon v-if="toast.type === 'error'" color="#F56C6C"><Close /></el-icon>
        <span>{{ toast.content }}</span>
      </div>
    </transition>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue'
import { Bell, Check, Warning, Close } from '@element-plus/icons-vue'
import ws from '../utils/websocket'
import { ElMessage } from 'element-plus'

const showPanel = ref(false)
const unreadCount = ref(0)
const notifications = ref<any[]>([])
const toast = ref<any>(null)
let toastTimer: number | null = null

const typeMap: Record<string, string> = {
  order_filled: 'success',
  order_rejected: 'danger',
  backtest_complete: 'primary',
  position_alert: 'warning',
  system: 'info'
}

const typeLabelMap: Record<string, string> = {
  order_filled: '成交',
  order_rejected: '拒绝',
  backtest_complete: '回测完成',
  position_alert: '持仓预警',
  system: '系统'
}

// 加载历史通知（从 localStorage，保留 24 小时）
const loadHistory = () => {
  try {
    const raw = localStorage.getItem('zq_notifications')
    if (!raw) return
    const all = JSON.parse(raw) as any[]
    const cutoff = Date.now() - 24 * 60 * 60 * 1000
    const valid = all.filter(n => n.time > cutoff)
    notifications.value = valid
    unreadCount.value = valid.filter(n => !n.read).length
  } catch { /* ignore */ }
}

// 保存到 localStorage
const saveNotifications = () => {
  localStorage.setItem('zq_notifications', JSON.stringify(notifications.value))
}

// 显示浮窗
const showToast = (content: string, type: string) => {
  toast.value = { content, type }
  if (toastTimer) clearTimeout(toastTimer)
  toastTimer = window.setTimeout(() => { toast.value = null }, 5000)
}

// 格式化时间
const formatTime = (time: string) => {
  const d = new Date(time)
  return `${d.getHours().toString().padStart(2, '0')}:${d.getMinutes().toString().padStart(2, '0')}:${d.getSeconds().toString().padStart(2, '0')}`
}

// 添加新通知
const addNotification = (n: any) => {
  const item = {
    id: Date.now().toString(),
    ...n,
    read: false,
    typeMap,
    typeLabel: typeLabelMap[n.type] || n.type
  }
  notifications.value.unshift(item)
  unreadCount.value++
  saveNotifications()
  showToast(n.content, n.type)
  // 播放提示音（若开启）
  const soundEnabled = localStorage.getItem('zq_sound_enabled') !== 'false'
  if (soundEnabled) {
    try {
      const audio = new Audio('/notification.mp3')
      audio.volume = 0.5
      audio.play().catch(() => {})
    } catch { /* ignore */ }
  }
}

// 点击通知
const onNotifClick = (n: any) => {
  n.read = true
  unreadCount.value = notifications.value.filter(x => !x.read).length
  saveNotifications()
  // 可跳转对应页面
}

// 清除全部
const clearAll = () => {
  notifications.value = []
  unreadCount.value = 0
  saveNotifications()
}

const togglePanel = () => {
  showPanel.value = !showPanel.value
}

onMounted(() => {
  loadHistory()

  // 监听 WebSocket 事件
  const eventTypes = ['order_update', 'position_update', 'backtest_result', 'system_alert']
  for (const type of eventTypes) {
    ws.on(type, (data: any) => {
      addNotification({
        type,
        content: typeof data === 'string' ? data : JSON.stringify(data).substring(0, 100),
        time: new Date().toISOString()
      })
    })
  }
})

onUnmounted(() => {
  if (toastTimer) clearTimeout(toastTimer)
  // 清理 ws 监听
})
</script>
```
