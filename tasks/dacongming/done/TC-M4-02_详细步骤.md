## Part A：持仓页面

### 步骤 A1：创建 `src/composables/usePosition.ts`

```typescript
import { ref, onMounted, onUnmounted } from 'vue'
import { backend as api } from '../api/backend'
import { data as dataApi } from '../api/data'
import ws from '../utils/websocket'

export interface PositionItem {
  id: number
  stockCode: string
  stockName: string
  quantity: number
  availableQty: number
  avgCost: number      // 成本价
  currentPrice: number // 当前价（实时）
  marketValue: number  // 市值
  profitLoss: number   // 盈亏金额
  profitRate: number   // 盈亏比例
}

export function usePosition() {
  const positions = ref<PositionItem[]>([])
  const loading = ref(false)

  // 加载持仓列表
  const fetchPositions = async () => {
    loading.value = true
    try {
      const res = await api.get('/position')
      positions.value = res.data || []

      // 逐个获取实时价格
      for (const pos of positions.value) {
        try {
          const quoteRes = await dataApi.get(`/market/${pos.stockCode}`)
          pos.currentPrice = quoteRes.data?.current_price || pos.avgCost
        } catch {
          pos.currentPrice = pos.avgCost // 获取失败则用成本价
        }
        // 计算盈亏
        pos.marketValue = pos.currentPrice * pos.quantity
        pos.profitLoss = (pos.currentPrice - pos.avgCost) * pos.quantity
        pos.profitRate = pos.avgCost > 0 ? pos.profitLoss / (pos.avgCost * pos.quantity) : 0
      }

      // 按盈亏比例降序排列
      positions.value.sort((a, b) => b.profitRate - a.profitRate)
    } catch (e) {
      console.error('fetchPositions failed', e)
    } finally {
      loading.value = false
    }
  }

  // 刷新单只股票价格（定时刷新）
  const refreshPrices = async () => {
    for (const pos of positions.value) {
      try {
        const quoteRes = await dataApi.get(`/market/${pos.stockCode}`)
        pos.currentPrice = quoteRes.data?.current_price || pos.currentPrice
        pos.marketValue = pos.currentPrice * pos.quantity
        pos.profitLoss = (pos.currentPrice - pos.avgCost) * pos.quantity
        pos.profitRate = pos.avgCost > 0 ? pos.profitLoss / (pos.avgCost * pos.quantity) : 0
      } catch { /* ignore */ }
    }
  }

  // WebSocket 实时更新
  onMounted(() => {
    fetchPositions()
    // 监听持仓更新事件
    ws.on('position_update', (data: any) => {
      positions.value = data
    })
  })

  onUnmounted(() => {
    ws.off('position_update')
  })

  return { positions, loading, fetchPositions, refreshPrices }
}
```

## Part B：订单列表页面

### 步骤 B1：创建 `src/composables/useOrders.ts`

```typescript
import { ref, computed } from 'vue'
import { backend as api } from '../api/backend'
import ws from '../utils/websocket'

export interface OrderItem {
  order_id: string
  created_at: string
  stock_code: string
  stock_name: string
  direction: number    // 1=buy, 2=sell
  price: number
  quantity: number
  filled_qty: number
  status: string      // pending/filled/cancelled/rejected
  commission?: number  // 手续费
}

const statusConfig: Record<string, { label: string; type: string }> = {
  pending:   { label: '待成交', type: 'warning' },
  filled:    { label: '已成交', type: 'success' },
  cancelled: { label: '已撤单', type: 'info' },
  rejected:  { label: '已拒绝', type: 'danger' }
}

export function useOrders() {
  const allOrders = ref<OrderItem[]>([])
  const loading = ref(false)
  const filterStatus = ref('')
  const filterDirection = ref('')
  const currentPage = ref(1)
  const pageSize = ref(20)

  const filteredOrders = computed(() => {
    let result = allOrders.value
    if (filterStatus.value) result = result.filter(o => o.status === filterStatus.value)
    if (filterDirection.value) result = result.filter(o => o.direction.toString() === filterDirection.value)
    return result
  })

  const paginatedOrders = computed(() => {
    const start = (currentPage.value - 1) * pageSize.value
    return filteredOrders.value.slice(start, start + pageSize.value)
  })

  const total = computed(() => filteredOrders.value.length)

  const fetchOrders = async () => {
    loading.value = true
    try {
      const res = await api.get('/order/list', { params: { page: 1, page_size: 100 } })
      allOrders.value = res.data || []
    } catch (e) {
      console.error('fetchOrders failed', e)
    } finally {
      loading.value = false
    }
  }

  // 撤单
  const cancelOrder = async (orderId: string) => {
    try {
      await api.post(`/order/cancel/${orderId}`)
      const order = allOrders.value.find(o => o.order_id === orderId)
      if (order) order.status = 'cancelled'
    } catch (e) {
      throw e
    }
  }

  // 获取订单详情（含成交明细）
  const getOrderDetail = async (orderId: string) => {
    const res = await api.get(`/order/${orderId}`)
    return res.data
  }

  // 撤单
  const cancelOrder = async (orderId: string) => {
    try {
      await api.post(`/order/cancel/${orderId}`)
      const order = allOrders.value.find(o => o.order_id === orderId)
      if (order) order.status = 'cancelled'
    } catch (e) {
      throw e
    }
  }

  onMounted(() => {
    fetchOrders()
    ws.on('order_update', (data: any) => {
      // 合并更新：新增或更新订单列表中的订单
      const idx = allOrders.value.findIndex(o => o.order_id === data.order_id)
      if (idx >= 0) {
        allOrders.value[idx] = { ...allOrders.value[idx], ...data }
      } else {
        allOrders.value.unshift(data)
      }
    })
  })

  onUnmounted(() => {
    ws.off('order_update')
  })

  return { allOrders, filteredOrders, paginatedOrders, total, loading, filterStatus, filterDirection, currentPage, pageSize, fetchOrders, cancelOrder, getOrderDetail, statusConfig }
}
```
