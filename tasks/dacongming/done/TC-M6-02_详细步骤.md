```typescript
  plugins: [
    vue(),
    AutoImport({
      resolvers: [VantResolver()],
      imports: ['vue', 'vue-router', 'pinia']
    }),
    Components({
      resolvers: [VantResolver()]
    }),
    gzipPlugin({
      algorithm: 'gzip',
      ext: '.gz'
    })
  ],
  resolve: {
    alias: {
      '@': resolve(__dirname, 'src')
    }
  },
  build: {
    target: 'es2015',
    minify: 'terser',
    rollupOptions: {
      output: {
        manualChunks: {
          'element-plus': ['element-plus'],
          'echarts': ['echarts'],
          'vant': ['vant'],
          'vendor': ['vue', 'vue-router', 'pinia']
        }
      }
    },
    chunkSizeWarningLimit: 1000 // KB
  },
  server: {
    port: 5173,
    proxy: {
      '/api': { target: 'http://localhost:8080', changeOrigin: true },
      '/data': { target: 'http://localhost:8080', changeOrigin: true }
    }
  }
})
```

## Phase 3：异常处理

### 步骤 3.1：创建全局请求错误处理 `src/utils/request.ts`
```typescript
import axios from 'axios'
import { ElMessage } from 'element-plus'

const request = axios.create({ baseURL: '/api/v1', timeout: 5000 })

let retryCount = 0
const maxRetry = 1

request.interceptors.response.use(
  response => response.data,
  async (error) => {
    const config = error.config
    if (retryCount < maxRetry && !config._retry) {
      retryCount++
      config._retry = true
      try {
        return await request(config)
      } catch { /* fall through */ }
    }
    retryCount = 0

    // 友好错误提示（不暴露技术细节）
    const msg = error.response?.data?.message || '网络请求失败，请稍后重试'
    ElMessage.error(msg)
    return Promise.reject(error)
  }
)

export default request
```

## Phase 5：Nginx 部署

### 步骤 5.1：创建部署目录
```bash
mkdir D:\ZeroQuant2\deploy\frontend
```
将 `dist/` 目录内容复制到 `D:\ZeroQuant2\deploy\frontend\`

### 步骤 5.2：创建 Nginx 配置 `D:\ZeroQuant2\deploy\nginx.conf`
```nginx
server {
    listen 80;
    server_name localhost;
    root D:/ZeroQuant2/deploy/frontend;
    index index.html;

    # SPA 路由支持
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API 代理到 Go 后端
    location /api/ {
        proxy_pass http://localhost:8080/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # 数据服务代理
    location /data/ {
        proxy_pass http://localhost:8080/data/;
        proxy_set_header Host $host;
    }

    # WebSocket 代理
    location /api/v1/ws {
        proxy_pass http://localhost:8080/api/v1/ws;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
    }

    # 静态资源缓存
    location /assets/ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # gzip 压缩
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;
    gzip_min_length 1000;
}
```
