## Phase 4：配置 API 封装（Axios）

### 步骤 4.1：创建后端 API 封装 `src/api/backend.ts`
```typescript
import axios from 'axios'
import type { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'

const backend: AxiosInstance = axios.create({
  baseURL: '/api/v1',
  timeout: 5000,
  headers: { 'Content-Type': 'application/json' }
})

// 请求拦截器：附加 JWT
backend.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('zq_token')
    if (token) config.headers.Authorization = `Bearer ${token}`
    return config
  },
  (error) => Promise.reject(error)
)

// 响应拦截器：统一处理 {code, message, data}
backend.interceptors.response.use(
  (response: AxiosResponse) => {
    const res = response.data
    if (res.code !== undefined && res.code !== 0) {
      return Promise.reject(new Error(res.message || '请求失败'))
    }
    return res
  },
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('zq_token')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

export default backend
```

## Phase 5：WebSocket 全局封装

### 步骤 5.1：创建 `src/utils/websocket.ts`
```typescript
import { ElMessage } from 'element-plus'

type WsCallback = (data: any) => void

class ZqWebSocket {
  private ws: WebSocket | null = null
  private url: string = ''
  private reconnectCount = 0
  private maxReconnect = 10
  private reconnectInterval = 2000
  private heartbeatTimer: number | null = null
  private listeners: Map<string, WsCallback[]> = new Map()

  connect(token: string) {
    const wsHost = import.meta.env.VITE_WS_URL || 'ws://localhost:8080/api/v1/ws'
    const finalUrl = wsHost + (wsHost.includes('?') ? '&' : '?') + `token=${token}`
    this.url = finalUrl
    this._connect()
  }

  private _connect() {
    try {
      this.ws = new WebSocket(this.url)
      this.ws.onopen = () => {
        console.log('[WS] Connected')
        this.reconnectCount = 0
        this._startHeartbeat()
      }
      this.ws.onmessage = (event) => {
        try {
          const msg = JSON.parse(event.data)
          const type = msg.type || 'message'
          const callbacks = this.listeners.get(type) || []
          callbacks.forEach(cb => cb(msg.data))
          // 通用回调
          const allCallbacks = this.listeners.get('*') || []
          allCallbacks.forEach(cb => cb(msg))
        } catch { /* ignore parse error */ }
      }
      this.ws.onclose = () => {
        console.warn('[WS] Closed, reconnecting...')
        this._stopHeartbeat()
        this._reconnect()
      }
      this.ws.onerror = (err) => {
        console.error('[WS] Error', err)
      }
    } catch (err) {
      console.error('[WS] Connect failed', err)
      this._reconnect()
    }
  }

  private _startHeartbeat() {
    this.heartbeatTimer = window.setInterval(() => {
      if (this.ws?.readyState === WebSocket.OPEN) {
        this.ws.send(JSON.stringify({ type: 'ping' }))
      }
    }, 10000) // 10秒心跳
  }

  private _stopHeartbeat() {
    if (this.heartbeatTimer) clearInterval(this.heartbeatTimer)
  }

  private _reconnect() {
    if (this.reconnectCount < this.maxReconnect) {
      this.reconnectCount++
      console.log(`[WS] Reconnecting ${this.reconnectCount}/${this.maxReconnect}`)
      setTimeout(() => this._connect(), this.reconnectInterval)
    } else {
      ElMessage.error('WebSocket 连接失败，请刷新页面重试')
    }
  }

  on(type: string, callback: WsCallback) {
    if (!this.listeners.has(type)) this.listeners.set(type, [])
    this.listeners.get(type)!.push(callback)
  }

  off(type: string, callback: WsCallback) {
    const cbs = this.listeners.get(type) || []
    this.listeners.set(type, cbs.filter(cb => cb !== callback))
  }

  close() {
    this._stopHeartbeat()
    this.ws?.close()
    this.ws = null
  }
}

// 全局单例
const ws = new ZqWebSocket()
export default ws
```

## Phase 6：路由配置（含权限守卫）

### 步骤 6.1：创建 `src/router/index.ts`
```typescript
import { createRouter, createWebHistory, type RouteRecordRaw } from 'vue-router'
import nprogress from 'nprogress' // 可选：加载进度条
import 'nprogress/nprogress.css'

const routes: RouteRecordRaw[] = [
  {
    path: '/login',
    name: 'Login',
    component: () => import('../views/Login.vue')
  },
  {
    path: '/',
    component: () => import('../layouts/MainLayout.vue'),
    redirect: '/dashboard',
    children: [
      { path: 'dashboard', name: 'Dashboard', component: () => import('../views/Dashboard.vue') },
      { path: 'strategy', name: 'Strategy', component: () => import('../views/Strategy.vue') },
      { path: 'strategy/editor', name: 'StrategyEditor', component: () => import('../views/StrategyEditor.vue') },
      { path: 'backtest', name: 'Backtest', component: () => import('../views/Backtest.vue') },
      { path: 'position', name: 'Position', component: () => import('../views/Position.vue') },
      { path: 'orders', name: 'Orders', component: () => import('../views/Orders.vue') },
    ]
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

// 路由守卫：未登录跳转 /login
router.beforeEach((to, from, next) => {
  nprogress?.start()
  const token = localStorage.getItem('zq_token')
  if (!token && to.path !== '/login') {
    next('/login')
  } else {
    next()
  }
})

router.afterEach(() => {
  nprogress?.done()
})

export default router
```

## Phase 7：Pinia 状态管理

### 步骤 7.1：创建 auth store `src/stores/auth.ts`
```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { backend } from '../api/backend'

export const useAuthStore = defineStore('auth', () => {
  const token = ref<string>(localStorage.getItem('zq_token') || '')
  const userInfo = ref<any>(null)

  const setToken = (t: string) => {
    token.value = t
    localStorage.setItem('zq_token', t)
  }

  const login = async (username: string, password: string) => {
    const res = await backend.post('/auth/login', { username, password })
    setToken(res.data.token)
    userInfo.value = res.data.user
    return res
  }

  const logout = () => {
    token.value = ''
    userInfo.value = null
    localStorage.removeItem('zq_token')
  }

  return { token, userInfo, setToken, login, logout }
})
```

<template>
  <div class="login-container">
    <el-card class="login-card">
      <h2>ZeroQuant 2.0 登录</h2>
      <el-form :model="form" @submit.prevent="handleLogin">

          <el-input v-model="form.username" placeholder="请输入用户名" />
        </el-form-item>

        </el-form-item>

          登录
        </el-button>
      </el-form>
    </el-card>
  </div>
</template>

<script setup lang="ts">
import { reactive, ref } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '../stores/auth'
import { ElMessage } from 'element-plus'

const router = useRouter()
const auth = useAuthStore()
const loading = ref(false)
const form = reactive({ username: '', password: '' })

  loading.value = true
  try {
    await auth.login(form.username, form.password)
    ElMessage.success('登录成功')
    router.push('/')
  } catch (e: any) {
    ElMessage.error(e.message || '登录失败')
  } finally {
    loading.value = false
  }
}
</script>
